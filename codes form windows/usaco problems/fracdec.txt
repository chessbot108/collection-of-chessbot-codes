int main() {
	int N, D, ind = 0;
	cin >> N >> D;
	
	oss << N/D << '.';
	
	N %= D;
	
	if(N == 0) oss << "0";
	
	while(vis.find(N) == vis.end()){
		vis[N] = ind++;
		dig.push_back(N);
		N = (N*10) % D;
	}
	
	
	
	for(int i = 0; i<vis[N]; i++){
		oss << (dig[i]*10) / D;
	}
	
	oss << "(";
	
	for(int i = vis[N]; i<dig.size(); i++){
		oss << (dig[i] * 10) / D;
	}
	
	oss << ")";
	
	string a = oss.str();
	string temp  (a.end() - 3, a.end());
	
	if(temp == "(0)"){ a.pop_back(); a.pop_back(); a.pop_back();}
	
	for(int i = 0; i<a.size(); i++){
		if(i && i%76 == 0) cout << "\n";
		cout << a[i];
	}
	
	
	cout << endl;
	
  	return 0;
}
